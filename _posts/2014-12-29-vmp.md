VMP 的基本逻辑运算只有一个或非, nor, 而不是什么与非

使用或非组合可以实现与或非三个运算

{% highlight python %}
def nor(v1, v2):
    return ~(v1 | v2)

def nor(v1, v2):
    return ~v1 & ~v2

def and(v1, v2):
    # ~(~v1 | ~v2) = v1 & v2
    return nor(nor(v1, v1), nor(v2, v2))

def or(v1, v2):
    # ~(~(v1 | v2)) = v1 | v2
    return nor(nor(v1, v2), nor(v1, v2))

{% endhighlight %}
